```go
package main

import (
	"fmt"
)

// 两个uint值相减，如果结果是负数会得到一个很大的uint值
// 负数在计算机中以原码的补码形式表示
// -5 的二进制表示:
// 10000000 00000000 00000000 00000101 原码，最高位为符号位，1表示负数
// 11111111 11111111 11111111 11111010 反码，除符号位外按位取反
// 11111111 11111111 11111111 11111011 补码
// 4 的二进制表示:
// 00000000 00000000 00000000 00000100
// 4-5 计算时为 4+(-5)，即：
// 00000000 00000000 00000000 00000100
// +
// 11111111 11111111 11111111 11111011
// =
// 11111111 11111111 11111111 11111111
// 转换成十进制uint32:
// uint32(math.Pow(2, 32) - 1)
// or
// strconv.ParseUint("11111111111111111111111111111111", 2, 64)
// return: 4294967295
func UintMinus() uint32 {
	var a uint32 = 4
	var b uint32 = 5
	return a - b
}

func main() {
	fmt.Println(UintMinus())
}
```
